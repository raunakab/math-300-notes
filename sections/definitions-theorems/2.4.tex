\begin{definition}[Cauchy-Riemann]
    \answer{
        A complex function, $f:\R\rightarrow\R$ is considered to satisfy Cauchy-Riemann iff:
        \begin{align}
            \bigg(u_x=v_y\bigg)\land\bigg(u_y=-v_x\bigg)
        \end{align}
        where $f:=u+iv$ and $u,v:\R^2\rightarrow\R$.
    }
\end{definition}
\begin{definition}[Differentiability and C-R]
    \answer{
        Consider $f:\C\rightarrow\C$ and some $z_0:\C$.
        \begin{align}
            \text{$f$ is differentiable at $z_0$}\implies\text{$f$ satisfies C-R at $z_0$}
        \end{align}
    }
\end{definition}
\begin{definition}[C-R and Differentiability]
    \answer{
        Similarly, consider $f:\C\rightarrow\C$ and some $z_0:\C$.
        \begin{align}
            \bigg(
                \text{$f_x,f_y$ exist and are continuous at $z_0$}
            \bigg)\land\bigg(
                \text{$f$ satisfies C-R at $z_0$}
            \bigg)\implies\text{$f$ is differentiable at $z_0$}
        \end{align}
    }
\end{definition}
